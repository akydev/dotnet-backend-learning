int		4 bytes 		Stores whole numbers from -2,147,483,648 to 2,147,483,647

long		8 bytes 		Stores whole numbers from -9,223,372,036,854,775,808 to 9,223,372,036,854,775,807

float		4 bytes			Stores fractional numbers, Sufficient for Storing 6 to 7 decimal digits

double		8 bytes			Stores fractional numbers, sufficient fro storing 15 decimal digits

bool 		1 bytes			Store true or false values

char		2 bytes			Stores a single characters/letters, surrounding by single quotes

string		2 bytes per character	Stores a sequence of characters, surrounded by doubles quotes
