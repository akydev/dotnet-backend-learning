namespace program
{
    public class MyClass
    {
        static void Main(string[] args)
        {
            Console.WriteLine("Please enter the year to find out it is a leap year or not!");
            int year = Convert.ToInt32(Console.ReadLine());


            if (year > 0)
            {
                if (year % 4 == 0)
                {
                    Console.WriteLine("The given year is a leap year!");
                }
                else
                {
                    Console.WriteLine("The given year is not a leap year!");
                }
            }

            else
            {
                Console.WriteLine("Please enter the valid year!");
            }
            
        }
    }
}
